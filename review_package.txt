    def _paginate(self, url: str, payload: Dict[str, Any]) -> List[Dict[str, Any]]:
        """Notion API 分頁處理器"""
        results = []
        cursor = None
        
        while True:
            if cursor:
                payload["start_cursor"] = cursor
            
            try:
                response = self._session.post(url, json=payload, timeout=self.DEFAULT_TIMEOUT)
                response.raise_for_status()
                data = response.json()
                
                results.extend(data.get("results", []))
                
                if not data.get("has_more"):
                    break
                
                cursor = data.get("next_cursor")
            except Exception as e:
                results.append({"error": f"Pagination failed: {str(e)}"})
                break
        
        return results
    
    def _extract_title(self, properties: Dict[str, Any]) -> str:
        """動態提取頁面標題"""
        for prop_data in properties.values():
            if prop_data.get("type") == "title":
                title_list = prop_data.get("title", [])
                if title_list:
                    return title_list[0].get("plain_text", "Untitled")
        return "Untitled"
    
    def _extract_status(self, properties: Dict[str, Any]) -> str:
        """動態提取頁面狀態"""
        for prop_data in properties.values():
            if prop_data.get("type") == "status":
                status_data = prop_data.get("status", {})
                return status_data.get("name", "Unknown")
        return "Unknown"
    
    def get_quiz_page(self) -> Optional[str]:
        """獲取「AI 中級應用規劃師題型練習」頁面 ID"""
        if not self.token:
            return None
        
        url = f"{self.base_url}/search"
        payload = {
            "query": "AI 中級應用規劃師題型練習",
            "filter": {"property": "object", "value": "page"}
        }
        
        try:
            response = self._session.post(url, json=payload, timeout=self.DEFAULT_TIMEOUT)
            response.raise_for_status()
            data = response.json()
            
            for result in data.get("results", []):
                if result.get("object") == "page":
                    title = self._extract_title(result.get("properties", {}))
                    if "AI 中級應用規劃師題型練習" in title:
                        return result["id"]
            
            return None
        except Exception as e:
            print(f"Error finding quiz page: {e}")
            return None
    
    def _get_heading_blocks(self, page_id: str) -> List[Dict[str, Any]]:
        """獲取頁面下所有 heading_2 區塊"""
        url = f"{self.base_url}/blocks/{page_id}/children"
        response = self._session.get(url, timeout=self.DEFAULT_TIMEOUT)
        response.raise_for_status()
        data = response.json()
        
        headings = []
        for block in data.get("results", []):
            if block.get("type") == "heading_2":
                heading_text = block.get("heading_2", {}).get("rich_text", [{}])[0].get("plain_text", "")
                headings.append({
                    "id": block["id"],
                    "text": heading_text,
                    "type": "heading_2"
                })
        
        return headings

        url = f"{self.base_url}/blocks/{heading_id}/children"
        response = self._session.get(url, timeout=self.DEFAULT_TIMEOUT)
        response.raise_for_status()
        data = response.json()
        
        questions = []
        for block in data.get("results", []):
            if block.get("type") == "heading_4':
                q_text = block.get("heading_4", {}).get("rich_text", [{}])[0].get("plain_text", "")
                questions.append({
                    "id": block["id"],
                    "question": q_text,
                    "type": "heading_4"
                })
        
        return questions
    
    def _get_question_details(self, question_id: str) -> Dict[str, Any]:
        """獲取題目詳細內容（選項、用戶答案、批改狀態）"""
        url = f"{self.base_url}/blocks/{question_id}/children"
        response = self._session.get(url, timeout=self.DEFAULT_TIMEOUT)
        response.raise_for_status()
        data = response.json()
        
        details = {
            "options": [],
            "user_answer": None,
            "robot_callout_id": None,
            "is_graded": False
        }
        
        for block in data.get("results", []):
            block_type = block.get("type")
            
            if block_type == "bulleted_list_item":
                text = block.get("bulleted_list_item", {}).get("rich_text", [{}])[0].get("plain_text", "")
                details["options"].append(text)
            
            elif block_type == "callout":
                icon = block.get("callout", {}).get("icon", {})
                emoji = icon.get("emoji", "")
                text = block.get("callout", {}).get("rich_text", [{}])[0].get("plain_text", "")
                
                if emoji == "✏️" and "我的答案：" in text:
                    answer = text.replace("我的答案：", "").strip()
                    if answer and answer != "(在此輸入內容)":
                        details["user_answer"] = answer
                
                elif emoji == "🤖":
                    details["robot_callout_id"] = block["id"]
                    if "等待批閱" not in text:
                        details["is_graded"] = True
        
        return details
    
    def check_pending_grading(self) -> List[Dict[str, Any]]:
        """
        巡檢 Notion，找出所有需要批改的試卷
        
        回傳格式：
        [
            {
                "heading_id": "xxx",
                "heading_text": "📅 AI 應用規劃師中級模擬練習 (2026-04-27 v1)",
                "questions": [
                    {
                        "id": "q1",
                        "question": "題目文字",
                        "user_answer": "A",
                        "robot_callout_id": "xxx"
                    },
                    ...
                ]
            },
            ...
        ]
        """
        quiz_page_id = self.get_quiz_page()
        if not quiz_page_id:
            print("未找到「AI 中級應用規劃師題型練習」頁面")
            return []
        
        # 獲取所有 heading_2（試卷）
        headings = self._get_heading_blocks(quiz_page_id)
        
        pending_quizzes = []
        
        for heading in headings:
            # 檢查是否為試卷標題
            if "AI 應用規劃師中級模擬練習" not in heading["text"]:
                continue
            
            # 獲取該試卷下的所有題目
            questions = self._get_question_blocks(heading["id"])
            
            # 檢查每個題目是否有用戶答案且尚未批改
            questions_to_grade = []
            for q in questions:
                details = self._get_question_details(q["id"])
                
                if details["user_answer"] and not details["is_graded"]:
                    questions_to_grade.append({
                        "id": q["id"],
                        "question": q["question"],
                        "user_answer": details["user_answer"],
                        "robot_callout_id": details["robot_callout_id"]
                    })
            
            # 如果有需要批改的題目，加入清單
            if questions_to_grade:
                pending_quizzes.append({
                    "heading_id": heading["id"],
                    "heading_text": heading["text"],
                    "questions": questions_to_grade
                })
        
        return pending_quizzes
    
    def grade_quiz(self, quiz_data: Dict[str, Any], difficulty: str = 'basic') -> Dict[str, Any]:
        """
        批改試卷並更新 Notion
        
        參數：
            quiz_data: check_pending_grading() 回傳的單個試卷資料
            difficulty: 試卷難度 ('basic' 或 'advanced')
        
        回傳：
            {
                "total": 10,
                "correct": 8,
                "score": 80,
                "details": [...]
            }
        """
        if difficulty not in self._question_bank:
            return {"error": f"Unknown difficulty: {difficulty}"}
        
        question_bank = self._question_bank[difficulty]
        
        # 建立題目映射（根據題目文字匹配）
        question_map = {}
        for i, q in enumerate(question_bank):
            # 使用題目前幾個字作為 key
            key = q['q'][:20]
            question_map[key] = {
                'index': i,
                'correct': q['correct'],
                'analysis': q['analysis']
            }
        
        # 批改結果
        results = {
            "total": len(quiz_data["questions"]),
            "correct": 0,
            "score": 0,
            "details": []
        }
        
        for i, q in enumerate(quiz_data["questions"]):
            user_answer = q["user_answer"]
            
            # 從題庫中匹配正確答案
            correct_answer = None
            analysis = None
            
            # 嘗試匹配題目
            for key, value in question_map.items():
                if key in q["question"]:
                    correct_answer = value['correct']
                    analysis = value['analysis']
                    break
            
            # 如果找不到匹配，使用預設值
            if not correct_answer:
                # 按順序匹配（假設題目順序相同）
                if i < len(question_bank):
                    correct_answer = question_bank[i]['correct']
                    analysis = question_bank[i]['analysis']
                else:
                    correct_answer = '?'
                    analysis = '題庫中找不到此題目'
            
            is_correct = user_answer.upper() == correct_answer.upper()
            
            if is_correct:
                results["correct"] += 1
            
            # 更新 Notion
            if q["robot_callout_id"]:
                status = "✅ 正確" if is_correct else f"❌ 錯誤（正確：{correct_answer}）"
                content = f"批閱結果：{status}\n\n正確解答：{correct_answer}\n\n解析：{analysis}"
                
                update_url = f"{self.base_url}/blocks/{q['robot_callout_id']}"
                payload = {
                    "callout": {
                        "rich_text": [
                            {
                                "type": "text",
                                "text": {"content": content}
                            }
                        ]
                    }
                }
                
                try:
                    update_response = self._session.patch(update_url, json=payload, timeout=self.DEFAULT_TIMEOUT)
                    update_response.raise_for_status()
                except Exception as e:
                    print(f"Failed to update question {i+1}: {e}")
            
            results["details"].append({
                "question": q["question"][:50],
                "user_answer": user_answer,
                "correct_answer": correct_answer,
                "is_correct": is_correct
            })
            
            # 避免 API 速率限制
            time.sleep(0.3)
        
        # 計算分數
        results["score"] = int((results["correct"] / results["total"]) * 100) if results["total"] > 0 else 0
        
        return results
    
    def generate_quiz(self, difficulty: str = 'basic') -> Dict[str, Any]:
        """
        生成新的試卷並添加到 Notion
        
        參數：
            difficulty: 試卷難度 ('basic' 或 'advanced')
        
        回傳：
            {
                "success": True,
                "heading_id": "xxx",
                "question_count": 10,
                "timestamp": "2026-04-27 12:00:00"
            }
        """
        if difficulty not in self._question_bank:
            return {"error": f"Unknown difficulty: {difficulty}", "success": False}
        
        quiz_page_id = self.get_quiz_page()
        if not quiz_page_id:
            return {"error": "Quiz page not found", "success": False}
        
        questions = self._question_bank[difficulty]
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M")
        
        # 創建試卷標題
        version = "進階版" if difficulty == 'advanced' else "基礎版"
        header_text = f"📅 AI 應用規劃師中級模擬練習 ({timestamp} {version})"
        
        header_payload = {
            "children": [
                {
                    "object": "block",
                    "type": "heading_2",
                    "heading_2": {
                        "rich_text": [{"type": "text", "text": {"content": header_text}}],
                        "is_toggleable": True
                    }
                }
            ]
        }
        
        url = f"{self.base_url}/blocks/{quiz_page_id}/children"
        
        try:
            # 創建標題
            response = self._session.patch(url, json=header_payload, timeout=self.DEFAULT_TIMEOUT)
            response.raise_for_status()
            heading_id = response.json()["results"][0]["id"]
            
            # 創建題目
            for i, q in enumerate(questions):
                # 創建 Toggle H4
                toggle_payload = {
                    "children": [
                        {
                            "object": "block",
                            "type": "heading_4",
                            "heading_4": {
                                "rich_text": [{"type": "text", "text": {"content": q['q']}}],
                                "is_toggleable": True
                            }
                        }
                    ]
                }
                
                toggle_response = self._session.patch(url, json=toggle_payload, timeout=self.DEFAULT_TIMEOUT)
                toggle_response.raise_for_status()
                toggle_id = toggle_response.json()["results"][0]["id"]
                
                # 創建選項和答案區塊
                children = []
                
                # 添加選項
                for option in q['options']:
                    children.append({
                        "object": "block",
                        "type": "bulleted_list_item",
                        "bulleted_list_item": {
                            "rich_text": [{"type": "text", "text": {"content": option}}]
                        }
                    })
                
                # 添加答案 callout
                children.append({
                    "object": "block",
                    "type": "callout",
                    "callout": {
                        "icon": {"type": "emoji", "emoji": "✏️"},
                        "rich_text": [{"type": "text", "text": {"content": "我的答案：\n(在此輸入內容)"}}]
                    }
                })
                
                # 添加批閱 callout
                children.append({
                    "object": "block",
                    "type": "callout",
                    "callout": {
                        "icon": {"type": "emoji", "emoji": "🤖"},
                        "rich_text": [{"type": "text", "text": {"content": "批閱結果：(等待批閱...)"}}]
                    }
                })
                
                # 提交子區塊
                child_response = self._session.patch(
                    f"{self.base_url}/blocks/{toggle_id}/children",
                    json={"children": children},
                    timeout=self.DEFAULT_TIMEOUT
                )
                child_response.raise_for_status()
                
                # 避免 API 速率限制
                time.sleep(0.5)
            
            return {
                "success": True,
                "heading_id": heading_id,
                "question_count": len(questions),
                "timestamp": timestamp,
                "difficulty": difficulty
            }
        
        except Exception as e:
            return {
                "success": False,
                "error": str(e)
            }
    
    def run_hourly_check(self) -> Dict[str, Any]:
        """
        每小時巡檢並自動批改
        
        回傳：
            {
                "checked_at": "2026-04-27 12:00:00",
                "quizzes_found": 2,
                "quizzes_graded": [
                    {
                        "heading_text": "...",
                        "score": 80,
                        "correct": 8,
                        "total": 10
                    }
                ]
            }
        """
        result = {
            "checked_at": datetime.now().strftime("%Y-%m-%d %H:%M:%S"),
            "quizzes_found": 0,
            "quizzes_graded": []
        }
        
        # 檢查需要批改的試卷
        pending_quizzes = self.check_pending_grading()
        result["quizzes_found"] = len(pending_quizzes)
        
        if not pending_quizzes:
            print(f"[{result['checked_at']}] 無需批改的試卷")
            return result
        
        # 批改每個試卷
        for quiz in pending_quizzes:
            # 判斷難度（從標題）
            difficulty = 'advanced' if '進階版' in quiz['heading_text'] else 'basic'
            
            print(f"\n正在批改：{quiz['heading_text']}")
            grading_result = self.grade_quiz(quiz, difficulty)
            
            if "error" not in grading_result:
                result["quizzes_graded"].append({
                    "heading_text": quiz["heading_text"],
                    "score": grading_result["score"],
                    "correct": grading_result["correct"],
                    "total": grading_result["total"]
                })
                print(f"  → 成績：{grading_result['correct']}/{grading_result['total']} ({grading_result['score']}分)")
            else:
                print(f"  → 批改失敗：{grading_result['error']}")
        
        return result

  python ai_quiz_scheduler.py --action=check
  
  # 完整測試（生成 + 檢查 + 批改）
  python ai_quiz_scheduler.py --action=all
        """
    )
    
    parser.add_argument(
        '--action',
        type=str,
        choices=['generate', 'grade', 'check', 'all'],
        default='check',
        help='執行動作：generate(生成試卷), grade(批改試卷), check(檢查), all(全部)'
    )
    
    parser.add_argument(
        '--difficulty',
        type=str,
        choices=['basic', 'advanced'],
        default=None,
        help='指定試卷難度（僅在 generate 動作時有效，預設依日期自動判斷）'
    )
    
    parser.add_argument(
        '--send-email',
        action='store_true',
        help='批改完成後發送 Email 通知'
    )
    
    args = parser.parse_args()
    
    if args.action == 'generate':
        difficulty = args.difficulty
        if not difficulty:
            # 自動判斷
            day = datetime.now().day
            difficulty = 'basic' if day % 2 == 1 else 'advanced'
        return daily_generate()
    
    elif args.action == 'grade':
        result = hourly_grade()
        if args.send_email and result['quizzes_graded']:
            send_email_notification(result)
        return result
    
    elif args.action == 'check':
        return manual_check()
    
    elif args.action == 'all':
        print("🔄 執行完整測試流程...\n")
        
        # 1. 生成試卷
        gen_result = daily_generate()
        print("\n等待 3 秒讓 Notion 同步...\n")
        time.sleep(3)
        
        # 2. 檢查
        check_result = manual_check()
        print("\n" + "="*60 + "\n")
        
        # 3. 批改
        grade_result = hourly_grade()
        
        return {
            'generate': gen_result,
            'check': check_result,
            'grade': grade_result
        }


if __name__ == "__main__":
    import time  # 用於 all 動作的延遲
    main()
